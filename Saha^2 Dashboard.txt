        let currentFilter = 'All';

        function calculatePilotState(user) {
            let avg = user.cycleLength.includes('-') ? (user.cycleLength.split('-').map(Number).reduce((a,b)=>a+b)/2) : 28;
            let dayOnLaunch = user.specificDayOnLaunch || (user.launchPhase === "Menstrual" ? 1 : user.launchPhase === "Follicular" ? 8 : user.launchPhase === "Luteal" ? 21 : 15);
            
            // Current Day = Day on Launch minus days remaining until launch
            let currentDay = dayOnLaunch - daysToLaunch;
            while (currentDay <= 0) currentDay += Math.round(avg);
            while (currentDay > Math.round(avg)) currentDay -= Math.round(avg);

            let currentPhase = PHASES.LUTEAL;
            if (currentDay <= 5) currentPhase = PHASES.MENSTRUAL;
            else if (currentDay <= 13) currentPhase = PHASES.FOLLICULAR;
            else if (currentDay <= 16) currentPhase = PHASES.OVULATORY;

            return { ...user, state: { currentDay, currentPhase, dayOnLaunch, launchPhase: user.launchPhase } };
        }

                </div>
            `).join('');

            // Intelligence
            const symptoms = {};
            stateData.forEach(u => u.symptoms.forEach(s => symptoms[s] = (symptoms[s] || 0) + 1));
            const top = Object.entries(symptoms).sort((a,b) => b[1]-a[1]).slice(0, 3);
            document.getElementById('intelligence-content').innerHTML = `

            `;

                        </div>
